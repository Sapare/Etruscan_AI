		}

Artillery_instead = {
		enable = {
			AND = {
				check_variable = { resource_produced@rubber < 2 }				
			NOT = {
			AND = {
				has_tech = basic_heavy_tank
				check_variable = { resource_produced@chromium > 12 }
			}
			}
			check_variable = { resource_produced@tungsten > 2 }
			}
			}
			
		abort_when_not_enabled = yes

			type = role_ratio

			value = -15

			value = -15
